def to_time_object
  return nil unless type == "TS"

  year = value[0,4].to_i
  month = value[4,2].to_i
  day = value[6,2].to_i
  hour = 0
  minute = 0
  second = 0
  if (value.length > 8)
    hour = value[8,2].to_i
    minute = value[10,2].to_i
    second = value[12,2].to_i
  end
  
  Time.gm(year, month, day, hour, minute, second)
end
